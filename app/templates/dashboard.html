Pipeline tweaks

      

        {# Embedded-subs short-circuit (0.10.0+). When ON, the pipeline
           checks the source for text subtitle tracks BEFORE running STT.
           Same-language tracks copied as-is; other-language text tracks
           skip STT and go straight to translation. Massively improves
           output quality and turnaround time on libraries with embedded
           subs. Default ON. #}
        {% if settings.prefer_embedded_subs %}
          embedded subs: on
        {% else %}
          embedded subs: off · STT always runs
        {% endif %}

        {# Vocal isolation — show the chosen mode + a small note that
           it auto-skips on 5.1+ sources. The footnote is the entire
           reason this got promoted onto the dashboard. #}
        {% set vi_mode = settings.vocal_isolation_mode %}
        {% if vi_mode == 'off' %}
          vocals: off
        {% elif vi_mode == 'chunked' %}
          vocals: chunked
          {{ settings.vocal_isolation_chunk_seconds }}s chunks
        {% else %}
          vocals: full · ≥ 12 GB RAM
        {% endif %}

        {# Readability polish — green when on (defaults), warn when
           explicitly disabled (likely a forgotten override). #}
        {% if settings.polish_enabled %}
          polish: on
        {% else %}
          polish: off · raw Whisper timing
        {% endif %}

        {# Openvino-only knob: VAD pre-filter. Hidden on cpu backend
